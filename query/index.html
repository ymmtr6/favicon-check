実装コード

function setFavicon(color) {
  const base = '/favicon-check/favicons/';
  const validColors = ['red','blue','green','yellow','purple','orange'];
  if (!validColors.includes(color)) return;

  // <link rel="icon"> を更新（なければ作成）
  let iconEl = document.querySelector('link[rel="icon"]');
  if (!iconEl) {
    iconEl = document.createElement('link');
    iconEl.rel = 'icon';
    document.head.appendChild(iconEl);
  }
  iconEl.type = 'image/png';
  iconEl.sizes = '32x32';
  iconEl.href = base + color + '-32.png';

  // SVG favicon
  let svgEl = document.querySelector('link[rel="icon"][type="image/svg+xml"]');
  if (!svgEl) {
    svgEl = document.createElement('link');
    svgEl.rel = 'icon';
    svgEl.type = 'image/svg+xml';
    document.head.appendChild(svgEl);
  }
  svgEl.href = base + color + '.svg';

  // apple-touch-icon
  let appleEl = document.querySelector('link[rel="apple-touch-icon"]');
  if (!appleEl) {
    appleEl = document.createElement('link');
    appleEl.rel = 'apple-touch-icon';
    document.head.appendChild(appleEl);
  }
  appleEl.sizes = '180x180';
  appleEl.href = base + color + '-180.png';

  // URLをpushState（ページ遷移なしでURL更新）
  const url = new URL(window.location);
  url.searchParams.set('color', color);
  history.pushState({color}, '', url);
}
注意： クエリパラメータによるfavicon変更は history.pushState を使用しているため、 ページリロードなしでURLが変わります。ただし、<meta property="og:image"> はSNSのクローラーが 取得するため、JSでの動的変更は SNS シェア時には反映されません。
